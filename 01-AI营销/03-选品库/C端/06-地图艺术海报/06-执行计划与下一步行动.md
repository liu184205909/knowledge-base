```javascript
// 前端编辑器逻辑（使用Product360 API或自定义React组件）
class MapEditor {
    constructor() {
        this.city = 'new-york';
        this.theme = 'minimalist';
        this.text = '';
        this.color = '#000000';
        this.previewDebounce = null;
    }

    // 实时预览（低分辨率，快速）
    async updatePreview() {
        clearTimeout(this.previewDebounce);
        this.previewDebounce = setTimeout(async () => {
            const preview = await fetch('/wp-json/maptoposter/v1/preview', {
                method: 'POST',
                body: JSON.stringify({
                    city: this.city,
                    theme: this.theme,
                    text: this.text,
                    color: this.color,
                    resolution: 'low' // 800x600，用于预览
                })
            });
            this.displayPreview(preview.url);
        }, 500); // 500ms防抖
    }

    // 高分辨率生成（付费后触发）
    async generateHighRes() {
        // 显示加载动画
        this.showLoading();

        // 调用生成API（高分辨率，可能需要30秒-2分钟）
        const result = await fetch('/wp-json/maptoposter/v1/generate', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'X-WP-Nonce': wpApiSettings.nonce
            },
            body: JSON.stringify({
                city: this.city,
                theme: this.theme,
                text: this.text,
                color: this.color,
                resolution: 'high', // 4000x3000，用于打印
                format: 'pdf' // 或 'jpg'
            })
        });

        // 返回下载链接或添加到WooCommerce订单
        return result.json();
    }
}

// 初始化编辑器
const editor = new MapEditor();
```
