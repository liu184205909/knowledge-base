#### Step 2: 配合自动化工具

**使用工具**: `02-自动化工具库/发布推广工具/02-邮件自动化.md`

**技术实现**:

```python
import sendgrid
from sendgrid.helpers.mail import *
import anthropic
import os

class EmailAutomationPipeline:
    """邮件自动化流水线"""

    def __init__(self):
        self.sendgrid_api_key = os.getenv("SENDGRID_API_KEY")
        self.anthropic_api_key = os.getenv("ANTHROPIC_API_KEY")
        self.sg = sendgrid.SendGridAPIClient(self.sendgrid_api_key)

    def generate_email_content(
        self,
        subscriber_type: str,
        content_type: str,
        topics: list
    ) -> dict:
        """使用Claude生成个性化邮件内容"""

        client = anthropic.Anthropic(api_key=self.anthropic_api_key)

        prompt = f"""生成邮件营销内容:

订阅者类型: {subscriber_type} (营销人员/创作者/小企业主)
内容类型: {content_type} (周报/教程/优惠)
主题: {topics}

请生成:
1. 主题行: 3个版本 (A/B测试)
2. 预览文本
3. 邮件正文 (300-500字,HTML格式)
4. 个性化标记: {{subscriber_type}}, {{subscriber_name}}
5. CTA按钮

返回JSON格式:
{{
  "subject_lines": ["版本1", "版本2", "版本3"],
  "preview_text": "预览文本",
  "body_html": "<html>...</html>",
  "body_text": "纯文本版本",
  "cta_text": "立即查看",
  "cta_url": "https://example.com"
}}"""

        message = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=2000,
            messages=[{"role": "user", "content": prompt}]
        )

        return json.loads(message.content[0].text)

    def send_email(
        self,
        to_email: str,
        subject: str,
        html_content: str,
        text_content: str,
        subscriber_data: dict
    ) -> bool:
        """发送邮件"""

        # 个性化内容
        html_content = html_content.replace(
            "{{subscriber_type}}",
            subscriber_data.get("type", "用户")
        ).replace(
            "{{subscriber_name}}",
            subscriber_data.get("name", "朋友")
        )

        # 创建邮件
        message = Mail(
            from_email="noreply@example.com",
            to_emails=to_email,
            subject=subject,
            html_content=html_content,
            plain_text_content=text_content
        )

        # 添加自定义追踪
        message.custom_args = {
            "subscriber_id": subscriber_data.get("id"),
            "email_type": subscriber_data.get("email_type")
        }

        try:
            response = self.sg.send(message)
            return True
        except Exception as e:
            print(f"邮件发送失败: {e}")
            return False

    def send_bulk_email(
        self,
        subscribers: list,
        email_content: dict
    ) -> dict:
        """批量发送邮件"""

        results = {
            "sent": 0,
            "failed": 0,
            "errors": []
        }

        for subscriber in subscribers:
            success = self.send_email(
                to_email=subscriber["email"],
                subject=email_content["subject_lines"][0],  # 使用版本1
                html_content=email_content["body_html"],
                text_content=email_content["body_text"],
                subscriber_data=subscriber
            )

            if success:
                results["sent"] += 1
            else:
                results["failed"] += 1
                results["errors"].append(subscriber["email"])

        return results

# 使用示例
pipeline = EmailAutomationPipeline()

# 生成邮件内容
content = pipeline.generate_email_content(
    subscriber_type="marketer",
    content_type="weekly_roundup",
    topics=["Jasper AI 2.0更新", "视频工具对比", "用户故事"]
)

# 发送给1000个营销人员订阅者
marketer_subscribers = get_subscribers_by_type("marketer", limit=1000)
results = pipeline.send_bulk_email(marketer_subscribers, content)

print(f"发送成功: {results['sent']}")
print(f"发送失败: {results['failed']}")
```
