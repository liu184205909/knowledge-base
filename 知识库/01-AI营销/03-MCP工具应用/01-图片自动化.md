## 方案B: Python脚本 (最灵活)

### 适合场景

- ✅ 集成到自动化流水线
- ✅ 有Python基础

### 2.1 环境准备

```bash

pip install openai pillow requests

# 设置API Key
export OPENAI_API_KEY="your-api-key"
```

### 2.2 Python脚本

创建 `generate_images.py`:

```python
import openai
import requests
import os
from pathlib import Path

# 初始化OpenAI客户端
openai.api_key = os.getenv("OPENAI_API_KEY")

def generate_image(prompt, filename, size="1792x1024"):
    """
    生成单张图片

    参数:
        prompt: 图片描述prompt
        filename: 保存文件名
        size: 图片尺寸 (默认16:9)
    """
    try:
        # 调用DALL-E 3 API
        response = openai.images.generate(
            model="dall-e-3",
            prompt=prompt,
            size=size,
            quality="standard",
            n=1,
        )

        # 获取图片URL
        image_url = response.data[0].url

        # 下载图片
        img_data = requests.get(image_url).content

        # 保存图片
        Path(filename).parent.mkdir(parents=True, exist_ok=True)
        with open(filename, 'wb') as f:
            f.write(img_data)

        print(f"✅ 已生成: {filename}")
        return True

    except Exception as e:
        print(f"❌ 生成失败: {e}")
        return False

def generate_article_images(title, summary, keywords, output_dir="./images"):
    """
    为文章生成5张配图

    参数:
        title: 文章标题
        summary: 文章摘要
        keywords: 关键词
        output_dir: 输出目录
    """
    # 定义5张图片的prompt
    prompts = [
        # 1. 封面图
        f"Professional cover image for: {title}. "
        f"Modern, clean business style. Keywords: {keywords}. "
        f"16:9 aspect ratio, high quality, minimal design.",

        # 2. 概念图
        f"Concept visualization: {summary[:200]}. "
        f"Minimalist, clean design, infographic style. "
        f"Professional, modern aesthetic.",

        # 3. 对比图
        f"Before and after comparison infographic for: {title}. "
        f"Side by side comparison, clear visual contrast. "
        f"Professional business style.",

        # 4. 流程图
        f"Step by step process diagram for: {keywords}. "
        f"Clear flowchart, numbered steps, modern design. "
        f"Professional infographic style.",

        # 5. 总结图
        f"Summary visual showing key takeaways from: {title}. "
        f"Clean layout, bullet points, modern design. "
        f"Professional business infographic."
    ]

    generated_files = []

    for i, prompt in enumerate(prompts, 1):
        filename = f"{output_dir}/{keywords.replace(' ', '_').replace(',', '')}_image_{i:02d}.png"
        if generate_image(prompt, filename):
            generated_files.append(filename)

    return generated_files

# 使用示例
if __name__ == "__main__":
    # 示例1: 生成单张图片
    generate_image(
        prompt="A smart pet health monitoring collar on a golden retriever, modern tech design, blue lighting",
        filename="product_cover.png"
    )

    # 示例2: 为文章生成5张图
    images = generate_article_images(
        title="AI写作工具完全指南",
        summary="本文将介绍10款顶级AI写作工具，帮助内容创作者提升效率",
        keywords="AI,writing,tools",
        output_dir="./images/article1"
    )

    print(f"\n✅ 总共生成 {len(images)} 张图片")
```

### 2.3 批量处理脚本

创建 `batch_generate.py`:

```python
import csv
from generate_images import generate_article_images

def batch_generate_from_csv(csv_file):
    """
    从CSV文件批量生成图片

    CSV格式:
    title,summary,keywords
    """
    with open(csv_file, 'r', encoding='utf-8') as f:
        reader = csv.DictReader(f)
        articles = list(reader)

    report = {
        "total": len(articles),
        "success": 0,
        "failed": 0,
        "total_images": 0
    }

    for idx, article in enumerate(articles, 1):
        print(f"\n[{idx}/{len(articles)}] 处理: {article['title']}")

        try:
            images = generate_article_images(
                title=article['title'],
                summary=article['summary'],
                keywords=article['keywords'],
                output_dir=f"./images/article_{idx}"
            )
            report['success'] += 1
            report['total_images'] += len(images)
        except Exception as e:
            print(f"❌ 失败: {e}")
            report['failed'] += 1

    print(f"\n{'='*50}")
    print(f"批量生成完成！")
    print(f"总文章数: {report['total']}")
    print(f"成功: {report['success']}")
    print(f"失败: {report['failed']}")
    print(f"总图片数: {report['total_images']}")

if __name__ == "__main__":
    batch_generate_from_csv("articles.csv")
```

### 2.4 运行脚本

```bash

python generate_images.py

# 批量处理
python batch_generate.py
```

```bash
npm install -g n8n

n8n start

# 访问: http://localhost:5678
```

### 3.2 创建工作流

**步骤1**: 添加触发器 (Trigger)
- 选择 "Cron" 或 "Webhook"
- 设置定时任务或手动触发

**步骤2**: 添加HTTP Request节点
- 配置OpenAI DALL-E API
- URL: `https://api.openai.com/v1/images/generations`
- Method: POST
- Headers:
  ```json
  {
    "Authorization": "Bearer YOUR_API_KEY",
    "Content-Type": "application/json"
  }
  ```
- Body:
  ```json
  {
    "model": "dall-e-3",
    "prompt": "{{$json.prompt}}",
    "size": "1792x1024",
    "quality": "standard",
    "n": 1
  }
  ```

**步骤3**: 添加保存图片节点
- 使用 "HTTP Request" 下载图片
- 使用 "Write Binary File" 保存到本地

**步骤4**: 测试工作流
- 点击 "Test workflow"
- 检查每个节点的输出

### 3.3 批量处理工作流

添加 "Loop Over Items" 节点:
- 输入: 文章列表 (JSON或CSV)
- 循环处理每篇文章
- 为每篇生成5张图

### 图片质量检查

```python
from PIL import Image

def check_image_quality(image_file):
    """检查图片质量"""
    img = Image.open(image_file)

    checks = {
        "尺寸": img.size,  # 应该是 (1792, 1024)
        "格式": img.format,  # 应该是 PNG
        "模式": img.mode,    # 应该是 RGB
        "文件大小": os.path.getsize(image_file) / 1024  # KB
    }

    return checks
```

### 图片压缩优化

```python
def optimize_image(image_file, quality=85):
    """压缩图片"""
    from PIL import Image

    img = Image.open(image_file)

    # 转换为RGB
    if img.mode != 'RGB':
        img = img.convert('RGB')

    # 压缩并保存
    optimized_file = image_file.replace('.png', '_optimized.jpg')
    img.save(optimized_file, 'JPEG', quality=quality, optimize=True)

    # 比较文件大小
    original_size = os.path.getsize(image_file) / 1024
    optimized_size = os.path.getsize(optimized_file) / 1024
    reduction = (1 - optimized_size/original_size) * 100

    print(f"压缩: {original_size:.1f}KB → {optimized_size:.1f}KB (减少{reduction:.1f}%)")

    return optimized_file
```
