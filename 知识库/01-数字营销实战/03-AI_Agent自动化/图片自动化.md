```python
# 在Claude Code中执行

import openai
import requests
import os

openai.api_key = "你的API_Key"

def generate_article_images(article_title, article_summary, keywords):
    """
    为文章自动生成5张配图

    参数:
        article_title: 文章标题
        article_summary: 文章摘要
        keywords: 关键词（逗号分隔）

    返回:
        生成的图片文件列表
    """

    # 定义5种图片类型和对应的prompt
    image_prompts = [
        # 图片1: 封面图
        f"Professional cover image for: {article_title}. "
        f"Modern, clean business style. Keywords: {keywords}. "
        f"16:9 aspect ratio, high quality, minimal design.",

        # 图片2: 概念图
        f"Concept visualization: {article_summary[:200]}. "
        f"Minimalist, clean design, infographic style. "
        f"Professional, modern aesthetic.",

        # 图片3: 对比图
        f"Before and after comparison infographic for: {article_title}. "
        f"Side by side comparison, clear visual contrast. "
        f"Professional business style.",

        # 图片4: 流程图
        f"Step by step process diagram for: {keywords}. "
        f"Clear flowchart, numbered steps, modern design. "
        f"Professional infographic style.",

        # 图片5: 总结图
        f"Summary visual showing key takeaways from: {article_title}. "
        f"Clean layout, bullet points, modern design. "
        f"Professional business infographic."
    ]

    generated_images = []

    for i, prompt in enumerate(image_prompts, 1):
        try:
            # 调用DALL-E 3 API
            response = openai.images.generate(
                model="dall-e-3",
                prompt=prompt,
                size="1792x1024",  # 16:9
                quality="standard",
                n=1,
            )

            image_url = response.data[0].url

            # 下载图片
            img_data = requests.get(image_url).content
            filename = f"{keywords.replace(' ', '_').replace(',', '')}_image_{i:02d}.png"

            with open(filename, 'wb') as f:
                f.write(img_data)

            generated_images.append(filename)
            print(f"✅ 已生成: {filename}")

        except Exception as e:
            print(f"❌ 生成失败 {i}: {e}")
            continue

    return generated_images

# 使用示例
article_title = "AI写作工具完全指南"
article_summary = "本文将介绍10款顶级AI写作工具，帮助内容创作者提升效率..."
keywords = "AI,writing,tools"

images = generate_article_images(article_title, article_summary, keywords)
print(f"\n✅ 总共生成 {len(images)} 张图片")
```

## Part 3: 批量生产流水线

### 3.1 完整自动化流程

```python
def batch_generate_images(articles_list):
    """
    批量生成文章配图

    参数:
        articles_list: 文章信息列表
        [
            {
                "title": "文章标题1",
                "summary": "摘要1",
                "keywords": "keyword1,keyword2"
            },
            ...
        ]

    返回:
        生成报告
    """

    report = {
        "total_articles": len(articles_list),
        "successful": 0,
        "failed": 0,
        "total_images": 0,
        "errors": []
    }

    for idx, article in enumerate(articles_list, 1):
        print(f"\n[{'='*50}]")
        print(f"处理文章 {idx}/{len(articles_list)}: {article['title']}")
        print(f"[{'='*50}]")

        try:
            # 生成图片
            images = generate_article_images(
                article['title'],
                article['summary'],
                article['keywords']
            )

            # 生成Alt标签
            alt_tags = generate_alt_tags(images, article['keywords'])

            # 保存Alt标签到文件
            with open(f"{article['keywords']}_alt_tags.txt", 'w') as f:
                for img, alt in alt_tags.items():
                    f.write(f"{img}: {alt}\n")

            report['successful'] += 1
            report['total_images'] += len(images)
            print(f"✅ 成功生成 {len(images)} 张图片 + Alt标签")

        except Exception as e:
            report['failed'] += 1
            report['errors'].append({
                "article": article['title'],
                "error": str(e)
            })
            print(f"❌ 失败: {e}")

    # 生成报告
    print(f"\n{'='*50}")
    print(f"批量生成完成！")
    print(f"{'='*50}")
    print(f"总文章数: {report['total_articles']}")
    print(f"成功: {report['successful']}")
    print(f"失败: {report['failed']}")
    print(f"总图片数: {report['total_images']}")

    if report['errors']:
        print(f"\n错误列表:")
        for error in report['errors']:
            print(f"  - {error['article']}: {error['error']}")

    return report

# 使用示例
articles = [
    {
        "title": "AI写作工具完全指南",
        "summary": "本文将介绍10款顶级AI写作工具...",
        "keywords": "AI,writing,tools"
    },
    {
        "title": "SEO优化实战技巧",
        "summary": "分享10个实用的SEO优化技巧...",
        "keywords": "SEO,optimization,tips"
    }
]

report = batch_generate_images(articles)
```

```python
def check_image_quality(image_file):
    """
    检查图片质量
    """
    from PIL import Image

    img = Image.open(image_file)

    checks = {
        "尺寸": img.size,  # 应该是 (1792, 1024)
        "格式": img.format,  # 应该是 PNG
        "模式": img.mode,    # 应该是 RGB
        "文件大小": os.path.getsize(image_file) / 1024  # KB
    }

    return checks
```

```python
def optimize_image(image_file, quality=85):
    """
    压缩图片以减少文件大小
    """
    from PIL import Image

    img = Image.open(image_file)

    # 转换为RGB（如果需要）
    if img.mode != 'RGB':
        img = img.convert('RGB')

    # 压缩并保存
    optimized_file = image_file.replace('.png', '_optimized.jpg')
    img.save(optimized_file, 'JPEG', quality=quality, optimize=True)

    # 比较文件大小
    original_size = os.path.getsize(image_file) / 1024
    optimized_size = os.path.getsize(optimized_file) / 1024
    reduction = (1 - optimized_size/original_size) * 100

    print(f"压缩完成: {original_size:.1f}KB → {optimized_size:.1f}KB (减少{reduction:.1f}%)")

    return optimized_file
```
