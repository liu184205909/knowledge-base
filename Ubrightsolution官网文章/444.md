// Main function: Download from Google Drive and return Media Library URL
function download_and_get_media_url($url, $filename = '') {
    if (empty($url)) {
        return '';
    }
 
    // Convert to direct download URL
    $download_url = convert_drive_to_direct($url);
 
    // Check if already in media library
    $existing_id = get_attachment_id_by_url($download_url);
    if ($existing_id) {
        $existing_url = wp_get_attachment_url($existing_id);
        if (!empty($existing_url)) {
            return $existing_url;
        }
    }
 
    // Check by filename if provided
    if (!empty($filename)) {
        $existing_id = get_attachment_id_by_filename($filename);
        if ($existing_id) {
            $existing_url = wp_get_attachment_url($existing_id);
            if (!empty($existing_url)) {
                return $existing_url;
            }
        }
    }
 
    // Download the file
    $attachment_id = download_to_media_library($download_url, $filename);
 
    if (!$attachment_id || is_wp_error($attachment_id)) {
        // If download fails, return the direct download URL
        return $download_url;
    }
 
    // Get the media library URL
    $media_url = wp_get_attachment_url($attachment_id);
 
    // Fallback to guid if wp_get_attachment_url fails
    if (empty($media_url)) {
        $attachment = get_post($attachment_id);
        if ($attachment && !empty($attachment->guid)) {
            $media_url = $attachment->guid;
        }
    }
 
    return !empty($media_url) ? $media_url : $download_url;
}
 
// Convert Google Drive URL to direct download
function convert_drive_to_direct($url) {
    if (empty($url)) {
        return '';
    }
 
    $url = trim($url);
 
    // Already direct link
    if (strpos($url, 'drive.google.com/uc?') !== false) {
        return $url;
    }
 
    // Extract file ID
    $patterns = [
        '/\/file\/d\/([a-zA-Z0-9-_]+)/',
        '/id=([a-zA-Z0-9-_]+)/',
        '/\/d\/([a-zA-Z0-9-_]+)/',
        '/open\?id=([a-zA-Z0-9-_]+)/'
    ];
 
    foreach ($patterns as $pattern) {
        if (preg_match($pattern, $url, $matches)) {
            return "https://drive.google.com/uc?export=download&id=" . $matches[1];
        }
    }
 
    return $url;
}
 
// Download and add to Media Library
function download_to_media_library($url, $filename = '') {
    if (!function_exists('media_handle_sideload')) {
        require_once(ABSPATH . 'wp-admin/includes/media.php');
        require_once(ABSPATH . 'wp-admin/includes/file.php');
        require_once(ABSPATH . 'wp-admin/includes/image.php');
    }
 
    // Increase timeout for large files
    add_filter('http_request_timeout', function() { return 300; });
 
    // Download to temp file
    $temp_file = download_url($url, 300);
 
    if (is_wp_error($temp_file)) {
        return false;
    }
 
    // Verify file exists and has content
    if (!file_exists($temp_file) || filesize($temp_file) == 0) {
        @unlink($temp_file);
        return false;
    }
 
    // Generate filename
    if (empty($filename)) {
        $filename = basename(parse_url($url, PHP_URL_PATH));
        $filename = preg_replace('/\?.*/', '', $filename);
 
        // If no filename, generate one
        if (empty($filename) || $filename == 'uc') {
            $filename = 'file-' . time();
        }
    }
 
    // Detect and add extension
    $filetype = wp_check_filetype($temp_file);
    $extension = '';
 
    if (!preg_match('/\.(mp3|wav|m4a|ogg|mp4|avi|mov|pdf|jpg|jpeg|png|gif|webm|flac)$/i', $filename)) {
        if (!empty($filetype['ext'])) {
            $extension = '.' . $filetype['ext'];
        } else {
            // Default to mp3 if can't detect
            $extension = '.mp3';
        }
        $filename .= $extension;
    }
 
    $filename = sanitize_file_name($filename);
 
    // Prepare file array
    $file_array = [
        'name'     => $filename,
        'tmp_name' => $temp_file,
        'error'    => 0,
        'size'     => filesize($temp_file),
    ];
 
    if (!empty($filetype['type'])) {
        $file_array['type'] = $filetype['type'];
    }
 
    // Upload to media library
    $attachment_id = media_handle_sideload($file_array, 0, null, [
        'test_form' => false,
    ]);
 
    // Cleanup
    if (file_exists($temp_file)) {
        @unlink($temp_file);
    }
 
    if (is_wp_error($attachment_id)) {
        return false;
    }
 
    return $attachment_id;
}
 
// Check if attachment exists by URL
function get_attachment_id_by_url($url) {
    global $wpdb;
 
    $attachment = $wpdb->get_var($wpdb->prepare(
        "SELECT ID FROM {$wpdb->posts} WHERE guid = %s AND post_type = 'attachment'",
        $url
    ));
 
    if ($attachment) {
        return intval($attachment);
    }
 
    // Check meta
    $attachment = $wpdb->get_var($wpdb->prepare(
        "SELECT post_id FROM {$wpdb->postmeta} WHERE meta_key = '_source_url' AND meta_value = %s",
        $url
    ));
 
    if ($attachment) {
        return intval($attachment);
    }
 
    return false;
}
 
// Check if attachment exists by filename
function get_attachment_id_by_filename($filename) {
    global $wpdb;
 
    $filename = sanitize_file_name($filename);
    $post_name = sanitize_title(pathinfo($filename, PATHINFO_FILENAME));
 
    $attachment = $wpdb->get_var($wpdb->prepare(
        "SELECT ID FROM {$wpdb->posts} WHERE post_name = %s AND post_type = 'attachment'",
        $post_name
    ));
 
    if ($attachment) {
        return intval($attachment);
    }
 
    return false;
}
 
// Alternative: Return attachment ID (for reference)
function download_and_get_attachment_id($url, $filename = '') {
    if (empty($url)) {
        return '';
    }
 
    $download_url = convert_drive_to_direct($url);
 
    $existing_id = get_attachment_id_by_url($download_url);
    if ($existing_id) {
        return $existing_id;
    }
 
    if (!empty($filename)) {
        $existing_id = get_attachment_id_by_filename($filename);
        if ($existing_id) {
            return $existing_id;
        }
    }
 
    $attachment_id = download_to_media_library($download_url, $filename);
    return $attachment_id ? $attachment_id : '';
}
